import threading
from database import init_db, search, get_stats
from indexer import Crawler
from search import perform_search

def display_help():
    print("\n--- CLI Commands ---")
    print("index <url> <depth>  - Start crawling from a URL")
    print("search <keyword>    - Search the database")
    print("status              - Show crawler and database stats")
    print("exit                - Quit application")

def main():
    # 1. Initialize Database
    print("[System] Initializing database...")
    init_db()

    # 2. Setup Crawler (Imported from indexer.py)
    crawler = Crawler(worker_count=5)
    crawler.start_background_loop()

    print("CLI System Ready. Type 'help' for commands.")

    # 3. Main Event Loop
    while True:
        try:
            cmd_input = input(">> ").strip().split()

                continue

                url, depth = cmd_input[1], int(cmd_input[2])
                crawler.add_job(url, depth)
                print(f"[System] Job queued: {url} (Depth: {depth})")

            elif command == "search":
                if len(cmd_input) < 2:
                    print("Usage: search <keyword>")
                    continue

                results = perform_search(query)
                print(f"Found {len(results)} results:")
                for res in results[:10]:
                    print(f" - [{res['score']}] {res['url']} | {res['title']}")

            elif command == "status":
                stats = get_stats()
                print(f"Pages: {stats['pages']} | Pending: {stats['pending']} | Workers: {stats['workers']}")

            elif command == "help":
                display_help()

            elif command == "exit":
                print("[System] Shutting down...")
                crawler.shutdown()
                break
            
            else:
                print("Unknown command. Type 'help'.")

        except KeyboardInterrupt:
            print("\n[System] Use 'exit' to quit.")
        except Exception as e:
            print(f"[Error] {e}")

if __name__ == "__main__":
    main()
```

### 3. Key Design Notes for `indexer.py`
To satisfy the requirements and the integration above, your `indexer.py` should implement the following structure:

*   **`class Indexer`**:
    *   `__init__`: Sets up a `threading.Thread` to run a `self.loop = asyncio.new_event_loop()`.
    *   `start_crawl(url, depth)`: Uses `self.loop.call_soon_threadsafe` to trigger the actual crawling coroutine.

2.  **Modularity**: `main.py` does not need to know how the crawler works (HTTP, parsing, etc.); it only needs to know the interface (`start_crawl`).
3.  **No Forbidden Libraries**: We used standard `threading` and `asyncio` for the concurrent orchestration, avoiding the need for Celery or Redis.

import asyncio
import sys
from database import Database
from indexer import Crawler
from search import SearchEngine

# ANSI Colors for Premium UX
class Colors:
    HEADER = '\033[95m'
    BLUE = '\033[94m'
    GREEN = '\033[92m'
    YELLOW = '\033[93m'
    RED = '\033[91m'
    END = '\033[0m'
    BOLD = '\033[1m'

def print_banner():
    banner = f"""
{Colors.BLUE}
   ___               _           _     ___               _           
  / __|_ _ __ _ __ _| |___ _ _  | |   / __|___ _ _  _ __| |___ _ _ 
 | (__| '_/ _` / _` | / -_) '_| | |__| (__/ _ \ ' \| '_ \ / -_) '_|
  \___|_| \__,_\__,_|_\___|_|   |____|\___\___/_||_| .__/_\___|_|  
                                                   |_|             
    {Colors.YELLOW}>> Advanced Asynchronous Crawler & Search Engine{Colors.END}
    """
    print(banner)

async def run_cli():
    db = Database("crawler.db")
    search_engine = SearchEngine("crawler.db")
    
    print_banner()
    print(f"{Colors.GREEN}System initialized. Database ready.{Colors.END}")

    while True:
        try:
            cmd = await asyncio.get_event_loop().run_in_executor(None, input, f"\n{Colors.BOLD}Crawler (User) > {Colors.END}")
            parts = cmd.split(maxsplit=1)
            if not parts: continue
            
            action = parts[0].lower()

            if action == "start":
                url = parts[1] if len(parts) > 1 else "https://example.com"
                print(f"{Colors.YELLOW}Starting crawl on: {url}{Colors.END}")
                crawler = Crawler(url, max_depth=2)
                # Run crawler in background
                asyncio.create_task(crawler.run())
                print(f"{Colors.GREEN}Crawler background task initiated.{Colors.END}")

            elif action == "search":
                if len(parts) < 2:
                    print(f"{Colors.RED}Please provide a search term.{Colors.END}")
                    continue
                query = parts[1]
                print(f"{Colors.BLUE}Searching for '{query}'...{Colors.END}")
                results = await search_engine.search(query)
                
                if not results:
                    print(f"{Colors.RED}No results found.{Colors.END}")
                else:
                    print(f"{'URL':<50} | {'Title':<20}")
                    print("-" * 75)
                    for r in results:
                        print(f"{r['url'][:50]:<50} | {r['title'][:20]:<20}")

            elif action == "help":
                print("Commands: start <url>, search <term>, exit")
            
            elif action == "exit":
                print("Shutting down...")
                break
                
        except Exception as e:
            print(f"{Colors.RED}Error: {e}{Colors.END}")

if __name__ == "__main__":
    try:
        asyncio.run(run_cli())
    except KeyboardInterrupt:
        sys.exit(0)
```

2.  **Concurrency (WAL Mode):** By leveraging the `database.py` and `search.py` modules you provided, this `main.py` respects the `WAL` mode settings. You can initiate a `start` command to crawl thousands of pages and immediately run a `search` command to query the database without waiting for the crawler to finish.
