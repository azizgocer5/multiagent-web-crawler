```python
        await db.execute("""CREATE TABLE IF NOT EXISTS Pages (

        super().__init__()
        self.links = set()
        self.base_url = base_url
    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            for attr, val in attrs:
                if attr == 'href':
                    self.links.add(urljoin(self.base_url, val))

        self.workers = workers
        self.queue = asyncio.Queue(maxsize=100)
        self.session = None

    async def worker(self, db_path):

            while True:
                item = await self.queue.get()
                if item is None: break
                url, depth = item
                
                try:
                    async with self.session.get(url, timeout=10) as resp:
                        html = await resp.text()
                        # Process content (Simplified)
                        await db.execute("INSERT OR IGNORE INTO Pages VALUES (?, ?, ?, ?)", 
                                         (url, "Title", "Body Snippet", depth))
                        
                        # Depth control logic
                        if depth < self.max_depth:
                            parser = LinkExtractor(url)
                            parser.feed(html)
                            for link in parser.links:
                                await db.execute("INSERT OR IGNORE INTO Queue (url, depth) VALUES (?, ?)", 
                                                 (link, depth + 1))
                        
                        await db.execute("UPDATE Queue SET state='done' WHERE url=?", (url,))
                        await db.commit()
                except Exception as e:
                    print(f"Failed {url}: {e}")
                finally:
                    self.queue.task_done()

    async def run(self, start_url, db_path="crawler.db"):
        await init_db(db_path)
        async with aiohttp.ClientSession() as self.session:
            # Seed initial URL
            async with aiosqlite.connect(db_path) as db:
                await db.execute("INSERT OR IGNORE INTO Queue (url, depth) VALUES (?, 0)", (start_url,))
                await db.commit()
                
                # Producer: Feed queue from DB
                tasks = [asyncio.create_task(self.worker(db_path)) for _ in range(self.workers)]
                
                cursor = await db.execute("SELECT url, depth FROM Queue WHERE state='pending'")
                async for row in cursor:
                    await self.queue.put(row)
                
                await self.queue.join()
                for _ in tasks: await self.queue.put(None)
                await asyncio.gather(*tasks)
```

```python
                    self.links.append(full_url)

    def handle_endtag(self, tag):
        if tag == 'title': self._in_title = False
        if tag in ('p', 'div', 'span'): self._in_body = False

    def handle_data(self, data):

        self.semaphore = asyncio.Semaphore(max_concurrent)
        self.queue = asyncio.Queue(maxsize=100)

    async def _fetch(self, session, url):
        async with self.semaphore:
            async with session.get(url, timeout=10) as response:

                await self._save_to_db(url, depth, parser)
            except Exception as e:
                logging.error(f"Failed to crawl {url}: {e}")
            finally:
                self.queue.task_done()

        parser.feed(html)
        return parser

            await db.commit()

            workers = [asyncio.create_task(self.worker(session, max_k)) for _ in range(5)]
            
            # Producer: Load pending from DB
            async with aiosqlite.connect(self.db_path) as db:
                async with db.execute("SELECT url, depth FROM Queue WHERE state='pending'") as cursor:
                    async for row in cursor:
                        await self.queue.put(row)
            
            await self.queue.join()
            for w in workers: w.cancel()
```

```python
import asyncio
import aiohttp
import sqlite3
import urllib.parse
from html.parser import HTMLParser

# --- Parser Logic ---
class LinkParser(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = set()
    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            for attr, value in attrs:
                if attr == 'href': self.links.add(value)

# --- Database Layer ---
class Database:
    def __init__(self, db_path="crawler.db"):
        self.conn = sqlite3.connect(db_path)
        self.conn.execute("PRAGMA journal_mode=WAL")
        self.conn.execute("CREATE TABLE IF NOT EXISTS pages (url TEXT UNIQUE, title TEXT, body TEXT, depth INTEGER)")
        self.conn.execute("CREATE TABLE IF NOT EXISTS queue (url TEXT UNIQUE, depth INTEGER, state TEXT)")
        self.conn.commit()

    def add_to_queue(self, url, depth, max_depth):
        if depth > max_depth: return
        try:
            self.conn.execute("INSERT INTO queue VALUES (?, ?, 'pending')", (url, depth))
            self.conn.commit()
        except sqlite3.IntegrityError: pass

# --- Crawler Engine ---
class Crawler:
    def __init__(self, start_url, max_depth, workers=10):
        self.db = Database()
        self.max_depth = max_depth
        self.queue = asyncio.Queue(maxsize=100)
        self.workers = workers
        self.db.add_to_queue(start_url, 0, max_depth)

    async def producer(self):
        while True:
            cursor = self.db.conn.execute("SELECT url, depth FROM queue WHERE state='pending' LIMIT 10")
            rows = cursor.fetchall()
            if not rows:
                await asyncio.sleep(1)
                continue
            for url, depth in rows:
                self.db.conn.execute("UPDATE queue SET state='processing' WHERE url=?", (url,))
                self.db.conn.commit()
                await self.queue.put((url, depth))
            await asyncio.sleep(0.1)

    async def worker(self):
        async with aiohttp.ClientSession() as session:
            while True:
                url, depth = await self.queue.get()
                try:
                    async with session.get(url, timeout=10) as response:
                        html = await response.text()
                        parser = LinkParser()
                        parser.feed(html)
                        
                        # Save content
                        self.db.conn.execute("INSERT OR IGNORE INTO pages VALUES (?, ?, ?, ?)", 
                                           (url, "Title", html[:500], depth))
                        self.db.conn.execute("UPDATE queue SET state='done' WHERE url=?", (url,))
                        
                        # Add children
                        for link in parser.links:
                            full_url = urllib.parse.urljoin(url, link)
                            self.db.add_to_queue(full_url, depth + 1, self.max_depth)
                        self.db.conn.commit()
                except Exception as e:
                    print(f"Error {url}: {e}")
                finally:
                    self.queue.task_done()

    async def run(self):
        tasks = [asyncio.create_task(self.producer())]
        for _ in range(self.workers):
            tasks.append(asyncio.create_task(self.worker()))
        await asyncio.gather(*tasks)

if __name__ == "__main__":
    crawler = Crawler("https://example.com", max_depth=2)
    asyncio.run(crawler.run())
```

```python
            await self.db.mark_as_pending(url, depth)

    async def _worker(self):
        """Worker process to pull from queue and crawl."""
        while True:
            url, depth = await self.queue.get()
            try:
                await self.process_page(url, depth)
            except Exception as e:
                logging.error(f"Error processing {url}: {e}")
            finally:
                self.queue.task_done()

    async def process_page(self, url, depth):
        """Downloads, parses, and persists page data using dict payload."""
        if depth > self.max_depth:
            return

        async with self.session.get(url, timeout=10) as response:
            if response.status != 200:
                return
            
            html = await response.text()
            parser = LinkParser()
            parser.feed(html)
            
            # Prepare data dictionary
            links = [urljoin(url, link) for link in parser.links]
            page_data = {
                "url": url,
                "title": parser.title,
                "body": " ".join(parser.body),
                "depth": depth,
                "links": links
            }
            
            # Persist via database helper using two arguments as required:
            # 1. The page metadata payload
            # 2. The explicit list of extracted links
            await self.db.save_page_and_links(page_data, links)

    async def run(self, start_url):
        """Main crawler loop."""
        async with aiohttp.ClientSession() as self.session:
            # Initialize: Add starting point
            await self.add_to_queue(start_url, 0)
            
            workers = [asyncio.create_task(self._worker()) for _ in range(self.max_workers)]
            
            while True:
                # Producer logic: fetch next batch from DB
                next_items = await self.db.get_pending_urls(limit=50)
                if not next_items:
                    # If workers are done and DB is empty, we finish
                    if self.queue.empty():
                        break
                    await asyncio.sleep(1)
                    continue
                
                for url, depth in next_items:
                    await self.queue.put((url, depth))
                
                # Backpressure: Wait until current batch is processed
                await self.queue.join()
```

```python
import asyncio
import aiohttp
import logging
from urllib.parse import urljoin
from html.parser import HTMLParser

class LinkParser(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = set()
        self.title = ""
        self.body = []
        self._in_title = False
        self._ignore_tags = {'script', 'style', 'noscript', 'meta', 'head'}
        self._current_tag = None

    def handle_starttag(self, tag, attrs):
        self._current_tag = tag
        if tag == 'a':
            for attr, value in attrs:
                if attr == 'href': self.links.add(value)
        elif tag == 'title': 
            self._in_title = True

    def handle_endtag(self, tag):
        if tag == 'title': self._in_title = False
        self._current_tag = None

    def handle_data(self, data):
        if self._current_tag in self._ignore_tags:
            return
        if self._in_title: 
            self.title += data
        else: 
            self.body.append(data.strip())

class Crawler:
    def __init__(self, db, max_depth=2, max_workers=30):
        self.db = db
        self.max_depth = max_depth
        self.max_workers = max_workers
        self.queue = asyncio.Queue(maxsize=100)
        self.session = None

    async def _worker(self):
        """Worker process to pull from queue and crawl."""
        while True:
            url, depth = await self.queue.get()
            try:
                await self.process_page(url, depth)
            except Exception as e:
                logging.error(f"Error processing {url}: {e}")
            finally:
                self.queue.task_done()

    async def process_page(self, url, depth):
        """Downloads, parses, and persists page data."""
        # 1. Depth Verification: Only process if within allowed limit
        if depth > self.max_depth:
            return

        async with self.session.get(url, timeout=10) as response:
            if response.status != 200:
                return
            
            html = await response.text()
            parser = LinkParser()
            parser.feed(html)
            
            # 2. Logic: Only schedule discovery if we haven't hit the max depth
            discovered_links = []
            if depth < self.max_depth:
                for link in parser.links:
                    full_url = urljoin(url, link)
                    # Add to DB immediately to prevent concurrent duplicate processing
                    if not await self.db.is_visited(full_url):
                        await self.db.mark_as_pending(full_url, depth + 1)
                        discovered_links.append(full_url)
            
            page_data = {
                "url": url,
                "title": parser.title.strip(),
                "body": " ".join(filter(None, parser.body)),
                "depth": depth,
            }
            
            # Persist data
            await self.db.save_page_and_links(page_data, discovered_links)
            # Mark the current URL as fully crawled in DB
            await self.db.mark_as_crawled(url)

    async def run(self, start_url):
        """Main crawler loop with DB state synchronization."""
        async with aiohttp.ClientSession() as self.session:
            # Seed the database
            if not await self.db.is_visited(start_url):
                await self.db.mark_as_pending(start_url, 0)
            
            workers = [asyncio.create_task(self._worker()) for _ in range(self.max_workers)]
            
            while True:
                # Producer logic: fetch batch from DB
                pending = await self.db.get_pending_urls(limit=50)
                
                if not pending:
                    if self.queue.empty():
                        break
                    await asyncio.sleep(1)
                    continue
                
                for url, depth in pending:
                    # Non-blocking put, only if queue isn't full
                    try:
                        await self.queue.put((url, depth))
                    except asyncio.QueueFull:
                        break
                
                await self.queue.join()
            
            # Cleanup
            for w in workers: w.cancel()
```

### Key Improvements:
*   **Decoupled Logic:** The `process_page` method now handles the `depth + 1` logic, ensuring that pages at `max_depth` do not spawn further child discovery.
*   **State Integrity:** By marking pages as `pending` in the database immediately upon discovery, we prevent multiple workers from picking up the same URL simultaneously.
*   **Robust Parsing:** Added a blacklist of tags (`script`, `style`, etc.) to the `LinkParser` to ensure the "body" text only contains human-readable content and not code/scripts.
